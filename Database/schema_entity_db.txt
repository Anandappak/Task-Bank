1) schema :- 
CREATE TABLE transaction_log (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    trx_id VARCHAR(100) NOT NULL,
    bank_id VARCHAR(50),
    customer_id BIGINT,
    from_account VARCHAR(50),
    to_account VARCHAR(50),
    amount DOUBLE,
    currency VARCHAR(10),
    status VARCHAR(30),
    created_at TIMESTAMP,

    CONSTRAINT uq_trx_id UNIQUE (trx_id)
);


2) index performance issue fetching
CREATE INDEX idx_customer_id ON transaction_log(customer_id);
CREATE INDEX idx_bank_id ON transaction_log(bank_id);
